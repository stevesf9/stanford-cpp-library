885 template <typename V, typename E>
886 std::ostream& operator <<(std::ostream& out, const VertexGen<V, E>& v) {
887  out << "Vertex{name=" << v.name;
888 
889 #ifdef SPL_BASICGRAPH_VERTEX_EDGE_RICH_MEMBERS
890  if (!floatingPointEqual(v.cost, 0.0)) {
891  out << ", cost=" << v.cost;
892  }
893  out << ", visited=" << (v.visited ? "true" : "false");
894  out << ", previous=" << (v.previous == nullptr ? string("nullptr") : v.previous->name);
895 #endif // SPL_BASICGRAPH_VERTEX_EDGE_RICH_MEMBERS
896 
897  // print neighbors
898  out << ", neighbors={";
899  int i = 0;
900  for (EdgeGen<V, E>* edge : v.edges) {
901  if (i > 0) {
902  out << ", ";
903  }
904  i++;
905  if (edge->finish) {
906  out << edge->finish->name;
907  } else {
908  out << "nullptr";
909  }
910  }
911  out << "}"; // end of neighbors
912 
913  out << "}"; // end of Vertex
914  return out;
915 }
