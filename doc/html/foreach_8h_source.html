21 /*
22  * Statement: foreach
23  * Usage: foreach (type var in collection) { ... }
24  * -----------------------------------------------
25  * The <code>foreach</code> statement steps through the elements in
26  * a collection. It works correctly with the collection classes in
27  * both the Standard Template Library and the Stanford C++ libraries,
28  * but can also be used with C++ strings and statically initialized
29  * arrays.
30  *
31  * <p>The following code, for example, prints every element in the
32  * string vector <code>lines</code>:
33  *
34  *<pre>
35  * foreach (string str in lines) {
36  * cout &lt;&lt; str &lt;&lt; endl;
37  * }
38  *</pre>
39  *
40  * Similarly, the following function calculates the sum of the character
41  * codes in a string:
42  *
43  *<pre>
44  * int sumCharacterCodes(string str) {
45  * int sum = 0;
46  * foreach (char ch in str) sum += ch;
47  * return sum;
48  * }
49  *</pre>
50  *
51  * As a simplification when iterating over maps, the <code>foreach</code>
52  * macro iterates through the keys rather than the key/value pairs.
53  */

127 template <typename DowncastType, typename ValueType>
128 ValueType HookImpl(State& fe) {
129  DowncastType *ip = (DowncastType *) fe.itr;
130  if (ip->iter == ip->end) {
131  fe.state = 2;
132  return ValueType();
133  }
134  fe.state = 1;
135  ValueType vp = *ip->iter; /* Subtle implementation note: */
136  ++ip->iter; /* Using *ip->iter++ here would */
137  return vp; /* require copying the iterator. */
138 }

167  fe.state = 2;
