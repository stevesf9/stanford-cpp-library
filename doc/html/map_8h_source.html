72  * Class: Map<KeyType,ValueType>
73  * -----------------------------
74  * This class maintains an association between <b><i>keys</i></b> and
75  * <b><i>values</i></b>. The types used for keys and values are
76  * specified using templates, which makes it possible to use
77  * this structure with any data type.
78  */
79 template <typename KeyType, typename ValueType>
80 class Map {
81 public:
82  /*
83  * Constructor: Map

87  * specified types.
88  */
89  Map();

91  /*
92  * Constructor: Map
93  * Usage: Map<KeyType,ValueType> map(lessFunc);
94  * --------------------------------------------
95  * Initializes a new empty map that associates keys and values of the
96  * specified types, using the given "less-than" comparison function
97  * to order any keys that will be later added to it.
98  * The function can accept the two keys to compare either by value
99  * or by const reference.
100  */
101  Map(std::function<bool (const KeyType&, const KeyType&)> lessFunc);
102 
103  /*
104  * Constructor: Map
105  * Usage: Map<ValueType> map {{"a", 1}, {"b", 2}, {"c", 3}};

109  * necessarily the order in which they are written in the initializer list.

387  * Returns the difference of the two maps, equivalent to a copy of the first map
388  * with removeAll called on it passing the second map as a parameter.
389  * You can also pass an initializer list of pairs such as {{"a", 1}, {"b", 2}, {"c", 3}}.

396  * --------------------

398  * Equivalent to calling removeAll(map2).
399  * You can also pass an initializer list of pairs such as {{"a", 1}, {"b", 2}, {"c", 3}}.
400  */
401  Map& operator -=(const Map& map2);

409  * You can also pass an initializer list of pairs such as {{"a", 1}, {"b", 2}, {"c", 3}}.
410  */
411  Map operator *(const Map& map2) const;
412

414  * Operator: *=
415  * Usage: map1 *= map2;
416  * ---------------------
417  * Removes all key/value pairs that are not found in the given map from this map.
418  * Equivalent to calling retainAll(map2).
419  * You can also pass an initializer list of pairs such as {{"a", 1}, {"b", 2}, {"c", 3}}.
420  */
421  Map& operator *=(const Map& map2);

581 template <typename KeyType, typename ValueType>
582 Map<KeyType, ValueType>& Map<KeyType, ValueType>::removeAll(const Map& map2) {
583  for (const KeyType& key : map2) {
584  if (containsKey(key) && get(key) == map2.get(key)) {
585  remove(key);
586  }
587  }
588  return *this;
589 }

591 template <typename KeyType, typename ValueType>
592 Map<KeyType, ValueType>& Map<KeyType, ValueType>::retainAll(const Map& map2) {
593  Vector<KeyType> toRemove;
594  for (const KeyType& key : *this) {
595  if (!map2.containsKey(key) || get(key) != map2.get(key)) {
596  toRemove.add(key);
597  }
598  }
599  for (const KeyType& key : toRemove) {
600  remove(key);
601  }
602  return *this;
603 }
