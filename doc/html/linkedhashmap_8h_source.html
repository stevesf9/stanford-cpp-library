565 template <typename KeyType, typename ValueType>
566 LinkedHashMap<KeyType, ValueType>& LinkedHashMap<KeyType, ValueType>::retainAll(const LinkedHashMap& map2) {
567  Vector<KeyType> toRemove;
568  for (const KeyType& key : *this) {
569  if (!map2.containsKey(key) || get(key) != map2.get(key)) {
570  toRemove.add(key);
571  }
572  }
573  for (const KeyType& key : toRemove) {
574  remove(key);
575  }
576  return *this;
577 }

648 // Implementation note:
649 // The definitions of <, <=, >, and >= are a bit unintuitive here.
650 // Because Maps are considered to be "equal" if they have the same key/value pairs,
651 // regardless of order, the equals() method and == / != operators ignore order.
652 //
653 // Similarly, if you ask whether a map is <= or >= to another, this includes
654 // the notion of equality, so it should return true if the maps contain the same
655 // pairs, regardless of order.
656 //
657 // If you ask whether a map is < or > to another, it is assumed that you don't
658 // want this to return true if they are "equal", so we must check for non-equality
659 // before checking the K/V pairs pairwise.
660 //
661 // This issue is unique to LinkedHashMap because Map sorts into a predictable sorted order
662 // and HashMap opts not to implement <, <=, >, or >= due to its unpredictable hash order.
663 
664 template <typename KeyType, typename ValueType>
665 bool LinkedHashMap<KeyType, ValueType>::operator <(const LinkedHashMap& map2) const {
666  return !equals(map2) && stanfordcpplib::collections::compareMaps(*this, map2) < 0;
667 }
