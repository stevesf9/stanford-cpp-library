28 /*
29  * An output stream buffer that counts the number of characters written to it.
30  * Used to limit cout console output by autograders for naughty students who
31  * print too much output.
32  *
33  * inspired by: http://gabisoft.free.fr/articles/fltrsbf1.html
34  */
35 class LimitOutputStreambuf : public std::streambuf {
36 public:
37  LimitOutputStreambuf(std::streambuf* source, int max) : m_source(source), m_count(0), m_max(max) {
38  // empty
39  }
40 
41  virtual ~LimitOutputStreambuf() {
42  sync();
43  }
44 
45  /*
46  * This is the crucial function; called to write a character to the
47  * underlying stream buffer (cout). We count them so we can throw
48  * an error if too many are printed.
49  */
50  virtual int overflow(int ch) {
51  if (m_count >= 0) {
52  m_count++;
53  if (m_count > m_max) {
54  m_count = -1; // disable checking on further calls so I can print again
55  std::ostringstream os;
56  os << std::endl;
57  os << "*** ERROR: Excessive output produced! (over " << m_max << " chars)" << std::endl;
58  os << "*** Halting program." << std::endl;
59  string str = os.str();
60  m_source->sputn(str.c_str(), (int) str.length());
61  throw std::exception();
62  }
63  }
64 
65  return m_source->sputc(ch);
66  }
67 
68  virtual int underflow() {
69  return EOF;
70  }
71 
72  virtual int sync() {
73  return m_source->pubsync();
74  }
75 
76  virtual std::streambuf* setbuf(char* p, std::streamsize len) {
77  return m_source->pubsetbuf(p, len);
78  }
79 
80 private:
81  std::streambuf* m_source;
82  int m_count;
83  int m_max;
84 };
