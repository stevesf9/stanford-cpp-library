708  class GraphComparator {
709  public:
710  bool operator ()(NodeType* n1, NodeType* n2) {
711  return compare(n1, n2) < 0;
712  }
713 
714  bool operator ()(ArcType* a1, ArcType* a2) {
715  return compare(a1, a2) < 0;
716  }
717  };

745  static int compare(NodeType* n1, NodeType* n2) {
746  if (n1 == n2) {
747  return 0;
748  }
749  if (n1->name < n2->name) {
750  return -1;
751  }
752  if (n1->name > n2->name) {
753  return +1;
754  }
755  return (n1 < n2) ? -1 : +1;
756  }
757 
763  static int compare(ArcType* a1, ArcType* a2) {
764  if (a1 == a2) {
765  return 0;
766  }
767  NodeType* n1 = a1->start;
768  NodeType* n2 = a2->start;
769  if (n1 != n2) {
770  return compare(n1, n2);
771  }
772  n1 = a1->finish;
773  n2 = a2->finish;
774  if (n1 != n2) {
775  return compare(n1, n2);
776  }
777  return (a1 < a2) ? -1 : +1;
778  }

859  * Implementation notes: addNode
860  * -----------------------------
861  * The addNode method appears in two forms: one that creates a node
862  * from its name and one that assumes that the client has created
863  * the new node. In each case, the implementation must add the node
864  * the set of nodes for the graph and add the name-to-node association
865  * to the node map.

867 template <typename NodeType, typename ArcType>
868 NodeType* Graph<NodeType, ArcType>::addNode(const string& name) {
869  NodeType* node = getNode(name);
870  if (node) {
871  return node; // vertex already exists
872  }
873  node = new NodeType();
874  node->arcs = Set<ArcType*>(GraphComparator());
875  node->name = name;
876  return addNode(node);

1455 template <typename NodeType, typename ArcType>
1456 int Graph<NodeType, ArcType>::graphCompare(const Graph<NodeType, ArcType>& graph2) const {
1457  // optimization: if literally the same graph, return true
1458  if (this == &graph2) {
1459  return 0;
1460  }
1461 
1462  auto itr1 = begin();
1463  auto itr2 = graph2.begin();
1464  auto g1end = end();
1465  auto g2end = graph2.end();
1466 
1467  while (itr1 != g1end && itr2 != g2end) {
1468  // compare each pair of elements from iterators
1469  NodeType* node1 = *itr1;
1470  NodeType* node2 = *itr2;
1471 
1472  // optimization: if literally same node, equal; don't compare
1473  if (node1 != node2) {
1474  // first check names
1475  if (node1->name != node2->name) {
1476  return node1->name.compare(node2->name);
1477  }
1478 
1479  // then check all arcs, pairwise
1480  auto eitr1 = node1->arcs.begin();
1481  auto eitr2 = node2->arcs.begin();
1482  auto e1end = node1->arcs.end();
1483  auto e2end = node2->arcs.end();
1484  while (eitr1 != e1end && eitr2 != e2end) {
1485  ArcType* arc1 = *eitr1;
1486  ArcType* arc2 = *eitr2;
1487 
1488  // optimization: if literally same arc, equal; don't compare
1489  if (arc1 != arc2) {
1490  // first check start vertex names, then end vertex names
1491  if (arc1->start->name != arc2->start->name) {
1492  return arc1->start->name.compare(arc2->start->name);
1493  } else if (arc1->finish->name != arc2->finish->name) {
1494  return arc1->finish->name.compare(arc2->finish->name);
1495  }
1496  }
1497  eitr1++;
1498  eitr2++;
1499  }
1500 
1501  // if we get here, everything from me matched graph2, so either arcs equal,
1502  // or one is shorter than the other (fewer arcs) and is therefore less
1503  if (eitr1 == e1end && eitr2 == e2end) {
1504  // keep going
1505  } else if (eitr1 == e1end) {
1506  return -1;
1507  } else {
1508  return 1;
1509  }
1510  }
1511 
1512  // if we get here, those two vertices and their outbound arcs
1513  // were equal; so advance to next element
1514  itr1++;
1515  itr2++;
1516  }
1517 
1518  // if we get here, everything from me matched graph2, so either equal,
1519  // or one is shorter than the other (fewer vertices) and is therefore less
1520  if (itr1 == g1end && itr2 == g2end) {
1521  return 0;
1522  } else if (itr1 == g1end) {
1523  return -1;
1524  } else {
1525  return 1;
1526  }
1527 }
