56  virtual int overflow(int ch, bool isStderr) {
57  string line = "";
58  for (char *cp = pbase(); cp < pptr(); cp++) {
59  if (*cp == '\n') {
60  myPutConsole(line, isStderr);
61  myEndLineConsole(isStderr);
62  line = "";
63  } else {
64  line += *cp;
65  }
66  }
67  if (line != "") {
68  myPutConsole(line, isStderr);
69  }
70  setp(outBuffer, outBuffer + BUFFER_SIZE);
71  if (ch != EOF) {
72  outBuffer[0] = ch;
73  pbump(1);
74  }
75  return ch != EOF;
76  }

79  return overflow();
80  }

83  return overflow(EOF, isStderr);
84  }

86  virtual int underflow() {
87  // Allow long strings at some point
88  blocked++;
89  string line = myGetLineConsole();
90  blocked--;
91 
92  bool eof = std::cin.eof();
93  fflush(stdout);
94 
95  if (eof) {
96  return EOF;
97  }
98 
99  int n = line.length();
100  if (n + 1 >= BUFFER_SIZE) {
101  error("ConsoleStreambuf::underflow: String too long");
102  }
103  for (int i = 0; i < n; i++) {
104  inBuffer[i] = line[i];
105  }
106  inBuffer[n++] = '\n';
107  inBuffer[n] = '\0';
108  setg(inBuffer, inBuffer, inBuffer + n);
109  return inBuffer[0];
110  }
111 };
