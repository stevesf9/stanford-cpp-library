167  static void checkAllocation(void* const p) {
168  unsigned long int stackDist, heapDist, staticDist;
169  stanfordcpplib::memory::computeMemoryDistances(p, stackDist, heapDist, staticDist);
170  if (stackDist < heapDist && stackDist < staticDist) {
171  if (!s_stackAllowed()) {
172  const std::type_info& type = typeid(T);
173  error(string("Stack allocation not allowed for type: ") + type.name()
174  + " (must allocate objects of this type using pointers and 'new')");
175  }
176  } else if (heapDist < stackDist && heapDist < staticDist) {
177  if (!s_heapAllowed()) {
178  const std::type_info& type = typeid(T);
179  error(string("Heap allocation not allowed for type: ") + type.name()
180  + " (must allocate objects of this type on the stack, without pointers or 'new')");
181  } else if (!s_heapWithoutNewAllowed() && !s_allocatedUsingNew().contains(p)) {
182  const std::type_info& type = typeid(T);
183  error(string("Invalid allocation for type: ") + type.name()
184  + " (must allocate objects of this type using pointers and 'new')");
185  }
186  } else {
187  if (!s_staticAllowed()) {
188  const std::type_info& type = typeid(T);
189  error(string("Static allocation not allowed for type: ") + type.name()
190  + " (must allocate objects of this type using pointers and 'new')");
191  }
192  }
