324  // optimization: if literally same collection, stop
325  if (&coll1 == &coll2) {
326  return true;
327  }
328  // optimization: if not same size, don't bother comparing pairwise
329  if (coll1.size() != coll2.size()) {
330  return false;
331  }
332 
333  // check each pair of elements for equality
334  auto itr1 = coll1.begin();
335  auto end1 = coll1.end();
336  auto itr2 = coll2.begin();
337  auto end2 = coll2.end();
338  while (itr1 != end1 && itr2 != end2) {
339  if (!(*itr1 == *itr2)) {
340  return false;
341  }
342  ++itr1;
343  ++itr2;
344  }
345  return true;
346 }

380  * key maps to the same value in both maps.
381  * The order of the keys is not considered.
382  * The key and value types must both have an operator ==.

384 template <typename MapType>
385 bool equalsMap(const MapType& map1, const MapType& map2) {
386  // optimization: if literally same map, stop
387  if (&map1 == &map2) {
388  return true;
389  }
390  // optimization: if not same size, don't bother comparing pairwise
391  if (map1.size() != map2.size()) {
392  return false;
393  }
394 
395  // check whether each element in the first map is also in the second.
396  // since the sizes are the same, if this is true, the maps are equal.
397  for (auto itr1 = map1.begin(), end1 = map1.end(); itr1 != end1; ++itr1) {
398  if (!map2.containsKey(*itr1) || !(map1.get(*itr1) == map2.get(*itr1))) {
399  return false;
400  }
401  }
402  return true;

406  * Computes a hashCode for any iterable type.
407  * Pass true for 'orderMatters' for collections that maintain some kind of element
408  * ordering that is of importance. This would be true for almost all collections
409  * except hash-based ones where the ordering is semi-random and unimportant.

412 int hashCodeIterable(IteratorType begin, IteratorType end, bool orderMatters = true) {
413  int code = hashSeed();
414  while (begin != end) {
415  if (orderMatters) {
416  code *= hashMultiplier();
417  }
418  code += hashCode(*begin);
419  ++begin;
420  }
421  return int(code & hashMask());
422 }

1911  "Oops! You tried using a type as a key in our Map without making it comparable. Click this error for more details.");
1912  /*
1913  * Hello CS106 students! If you got directed to this line of code in a compiler error,
1914  * it probably means that you tried making a Map with a custom struct or class type
1915  * as the key type or a Set with a custom struct as a value type.
1916  *
1917  * In order to have a type be a key type in a Map - or to have a type be a value type
1918  * in a Set - it needs to have be comparable using the < operator. By default, types in C++
1919  * can't be compared using the < operator, hence the error.
1920  *
1921  * There are two ways to fix this. The first option would simply be to not use your custom
1922  * type as a key in the Map or value in a Set. This is probably the easiest option.
1923  *
1924  * The second way to fix this is to explicitly define an operator< function for your custom
1925  * type. Here's the syntax for doing that:
1926  *
1927  * bool operator< (const YourCustomType& lhs, const YourCustomType& rhs) {
1928  * return compareTo(lhs.data1, rhs.data1,
1929  * lhs.data2, rhs.data2,
1930  * ...
1931  * lhs.dataN, rhs.dataN);
1932  * }
1933  *
1934  * where data1, data2, ... dataN are the data members of your type. For example, if you had
1935  * a custom type
1936  *
1937  * struct MyType {
1938  * int myInt;
1939  * string myString;
1940  * };
1941  *
1942  * you would define the function
1943  *
1944  * bool operator< (const MyType& lhs, const MyType& rhs) {
1945  * return compareTo(lhs.myInt, rhs.myInt,
1946  * lhs.myString, rhs.myString);
1947  * }
1948  *
1949  * Hope this helps!
1950  */
1951  return std::less<T>();
1952 }
