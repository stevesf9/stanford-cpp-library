76  * This class stores an ordered list of values similar to an array.

645 template <typename ValueType>
646 int LinkedList<ValueType>::lastIndexOf(const ValueType& value) const {
647  // loop using iterator to avoid O(N^2) runtime
648  int i = size();
649  auto begin = this->begin();
650  auto itr = this->end();
651  while (itr != begin) {
652  itr--;
653  i--;
654  if (*itr == value) {
655  return i;
656  }
657  }
658  return -1;
659 }
