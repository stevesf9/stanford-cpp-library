3  * ------------------
4  * This file exports the <code>SparseGrid</code> class, which offers a
5  * convenient abstraction for representing a two-dimensional array.
6  * Unlike Grid, SparseGrid is optimized for grids in which most of the cells
7  * are empty. It uses far less memory in such cases than a Grid does.
8  * If the grid is expected to be mostly full of meaningful data,
9  * Grid is recommended for use over SparseGrid.
10  *

414  * Implementation notes: SparseGrid data structure
415  * -----------------------------------------------
416  * The SparseGrid is internally managed as a dynamic array of elements.
417  * The array itself is one-dimensional, the logical separation into
418  * rows and columns is done by arithmetic computation. The layout
419  * is in row-major order, which is to say that the entire first row
420  * is laid out contiguously, followed by the entire second row,
421  * and so on.

648 template <typename ValueType>
649 SparseGrid<ValueType>::SparseGrid(std::initializer_list<std::initializer_list<ValueType>> list)
650  : _rowCount(0),
651  _columnCount(0) {
652  // create the grid at the proper size
653  _rowCount = list.size();
654  if (list.begin() != list.end()) {
655  _columnCount = list.begin()->size();
656  }
657  resize(_rowCount, _columnCount);
658 
659  // copy the data from the initializer list into the Grid
660  auto rowItr = list.begin();
661  for (int row = 0; row < _rowCount; row++) {
662  if ((int) rowItr->size() != _columnCount) {
663  error("SparseGrid::constructor: initializer list is not rectangular (must have same # cols in each row)");
664  }
665  auto colItr = rowItr->begin();
666  for (int col = 0; col < _columnCount; col++) {
667  set(row, col, *colItr);
668  colItr++;
669  }
670  rowItr++;
671  }
672 }

695 template <typename ValueType>
696 bool SparseGrid<ValueType>::equals(const SparseGrid<ValueType>& grid2) const {
697  // optimization: if literally same grid, stop
698  if (this == &grid2) {
699  return true;
700  }
701  if (_rowCount != grid2._rowCount || _columnCount != grid2._columnCount) {
702  return false;
703  }
704  for (int row = 0; row < _rowCount; row++) {
705  for (int col = 0; col < _columnCount; col++) {
706  if (isSet(row, col)) {
707  // I have data there; he must, too, and it must be the same data
708  if (!grid2.isSet(row, col) || get(row, col) != grid2.get(row, col)) {
709  return false;
710  }
711  } else {
712  // I don't have data there; he must also not have it there
713  if (grid2.isSet(row, col)) {
714  return false;
715  }
716  }
717  }
718  }
719  return true;
720 }

842 void SparseGrid<ValueType>::resize(int nRows, int nCols, bool retain) {
843  if (nRows < 0 || nCols < 0) {
844  std::ostringstream out;
845  out << "SparseGrid::resize: Attempt to resize grid to invalid size ("
846  << nRows << ", " << nCols << ")";
847  error(out.str());
848  }
849  int oldnRows = this->_rowCount;
850  int oldnCols = this->_columnCount;
851  this->_rowCount = nRows;
852  this->_columnCount = nCols;
853 
854  if (retain) {
855  // if resizing to a smaller size, must evict any row/col entries
856  // that exceed the new grid's bounds
857  if (nRows < oldnRows || nCols < oldnCols) {
858  Map<int, Map<int, ValueType>> newElements;
859  for (int row : _elements) {
860  if (row >= nRows) {
861  break; // don't add any entries beyond this row
862  }
863  for (int col : _elements[row]) {
864  if (col >= nCols) {
865  break; // don't add any entries beyond this row
866  }
867  newElements[row][col] = _elements[row][col];
868  }
869  }
870  _elements = newElements;
871  }
872  } else {
873  _elements.clear();
874  }
875  _version++;
876 }

984 template <typename ValueType>
985 int SparseGrid<ValueType>::gridCompare(const SparseGrid& grid2) const {
986  int h1 = height();
987  int w1 = width();
988  int h2 = grid2.height();
989  int w2 = grid2.width();
990  int rows = h1 > h2 ? h1 : h2;
991  int cols = w1 > w2 ? w1 : w2;
992  for (int r = 0; r < rows; r++) {
993  for (int c = 0; c < cols; c++) {
994  if (r >= h1) {
995  return -1;
996  } else if (r >= h2) {
997  return 1;
998  }
999 
1000  if (c >= w1) {
1001  return -1;
1002  } else if (c >= w2) {
1003  return 1;
1004  }
1005 
1006  if (!isSet(r, c) && grid2.isSet(r, c)) {
1007  return -1;
1008  } else if (isSet(r, c) && !grid2.isSet(r, c)) {
1009  return 1;
1010  }
1011 
1012  if (get(r, c) < grid2.get(r, c)) {
1013  return -1;
1014  } else if (grid2.get(r, c) < get(r, c)) {
1015  return 1;
1016  }
1017  }
1018  }
1019  return 0;
