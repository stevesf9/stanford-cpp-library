471  struct TrieNode {
472  public:
473  TrieNode() {
474  _isWord = false;
475  for (int i = 0; i < 26; i++) {
476  _children[i] = nullptr;
477  }
478  }
479 
480  // pre: letter is between 'a' and 'z' in lowercase
481  inline TrieNode*& child(char letter) {
482  return _children[letter - 'a'];
483  }
484 
485  inline int childCount() const {
486  int count = 0;
487  for (int i = 0; i < 26; i++) {
488  if (_children[i] != nullptr) {
489  count++;
490  }
491  }
492  return count;
493  }
